# Filtro por rango de edad
GET /employees/_search
{
  "query":{ 
    "bool":{
      "filter":{
        "range": {
          "birthdate": {"gte": "1998-02-05","lte": "2001-02-05"}
        }
      }
    }
  }
}

# Filtro inverso
GET employees/_search
{
  "query": {
    "bool": {
     "must_not": [
        {
          "range": {
            "birthdate": {"gte": "1998-02-05","lte": "2002-02-05"}
          }
        }
      ]
    }
  }
}
